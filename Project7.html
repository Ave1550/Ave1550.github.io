# Line plot with customized style

plt.plot(df['Date'], df['Price'], color='royalblue', linewidth=2, label='Brent Oil Price')
plt.title('Brent Crude Oil Price Over Time', fontsize=16)
plt.xlabel('Date', fontsize=12)
plt.ylabel('Price (USD)', fontsize=12)
plt.grid(True, alpha=0.3)
